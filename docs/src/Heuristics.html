frequency ps v = length $ filter (== v) $ foldl' (\acc p -> (vars p) ++ acc) [] ps

frequencies ps vs = foldl' (\acc v -> (v,frequency ps v):acc) [] vs

heuristics ps vs = foldr (\x acc -> ((fst x):acc)) [] $ sortOn snd $ frequencies ps vs
